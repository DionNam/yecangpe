```typescript
import { transliterate } from 'transliteration'

/**
 * Generate URL-friendly slug from job title.
 * Handles Korean characters via romanization.
 * Always appends UUID prefix for uniqueness (Korean titles can romanize identically).
 */
export function generateJobSlug(title: string, id: string): string {
  // Romanize Korean and other non-Latin characters
  const romanized = transliterate(title)

  // Slugify: lowercase, replace non-alphanumeric with hyphens, collapse multiple hyphens
  const slug = romanized
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-+|-+$/g, '')  // trim leading/trailing hyphens
    .replace(/-{2,}/g, '-')   // collapse multiple hyphens

  // Append first 8 chars of ID for uniqueness
  const suffix = id.replace(/-/g, '').slice(0, 8)

  return slug ? `${slug}-${suffix}` : suffix
}
```
