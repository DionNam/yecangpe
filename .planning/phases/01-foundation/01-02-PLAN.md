<task type="auto">
  <name>Task 1: Initialize monorepo structure with root configuration</name>
  <files>package.json, pnpm-workspace.yaml, turbo.json, .gitignore, .npmrc, packages/typescript-config/package.json, packages/typescript-config/base.json, packages/typescript-config/nextjs.json</files>
  <action>
    1. Create root package.json:
    ```json
    {
      "name": "hire-foreigner",
      "private": true,
      "scripts": {
        "dev": "turbo dev",
        "build": "turbo build",
        "lint": "turbo lint",
        "type-check": "turbo type-check",
        "db:gen-types": "turbo db:gen-types"
      },
      "devDependencies": {
        "turbo": "^2.4.0",
        "typescript": "^5.7.0"
      },
      "packageManager": "pnpm@9.15.0"
    }
    ```

    2. Create pnpm-workspace.yaml:
    ```yaml
    packages:
      - 'apps/*'
      - 'packages/*'
    ```

    3. Create turbo.json (from research):
    ```json
    {
      "$schema": "https://turbo.build/schema.json",
      "tasks": {
        "build": {
          "dependsOn": ["^build"],
          "outputs": [".next/**", "!.next/cache/**", "dist/**"]
        },
        "dev": {
          "cache": false,
          "persistent": true
        },
        "lint": {
          "dependsOn": ["^build"]
        },
        "type-check": {
          "dependsOn": ["^build"]
        },
        "db:gen-types": {
          "cache": false
        }
      }
    }
    ```

    4. Create .npmrc:
    ```
    auto-install-peers=true
    strict-peer-dependencies=false
    ```

    5. Create packages/typescript-config/package.json:
    ```json
    {
      "name": "@repo/typescript-config",
      "version": "0.0.0",
      "private": true,
      "files": ["*.json"]
    }
    ```

    6. Create packages/typescript-config/base.json:
    ```json
    {
      "$schema": "https://json-schema.org/draft/2020-12/schema",
      "compilerOptions": {
        "strict": true,
        "esModuleInterop": true,
        "skipLibCheck": true,
        "forceConsistentCasingInFileNames": true,
        "moduleResolution": "bundler",
        "module": "ESNext",
        "target": "ES2022",
        "lib": ["ES2022"],
        "declaration": true,
        "declarationMap": true,
        "sourceMap": true,
        "isolatedModules": true
      }
    }
    ```

    7. Create packages/typescript-config/nextjs.json:
    ```json
    {
      "$schema": "https://json-schema.org/draft/2020-12/schema",
      "extends": "./base.json",
      "compilerOptions": {
        "lib": ["dom", "dom.iterable", "ES2022"],
        "module": "ESNext",
        "moduleResolution": "bundler",
        "allowJs": true,
        "jsx": "preserve",
        "noEmit": true,
        "incremental": true,
        "plugins": [{ "name": "next" }]
      }
    }
    ```

    8. Update .gitignore to include:
    - node_modules
    - .next
    - dist
    - .turbo
    - .env*.local
    - *.tsbuildinfo
  </action>
  <verify>
    - All files created in root and packages/typescript-config
    - `pnpm install` runs without errors (will install turbo)
  </verify>
  <done>Root monorepo configuration complete with Turborepo and pnpm workspaces</done>
</task>

<task type="auto">
  <name>Task 2: Create apps/web and apps/admin Next.js applications</name>
  <files>apps/web/*, apps/admin/*</files>
  <action>
    For BOTH apps/web AND apps/admin:

    1. Create package.json:
    ```json
    {
      "name": "@repo/web",  // or "@repo/admin" for admin
      "version": "0.0.0",
      "private": true,
      "scripts": {
        "dev": "next dev --port 3000",  // or 3001 for admin
        "build": "next build",
        "start": "next start",
        "lint": "next lint",
        "type-check": "tsc --noEmit"
      },
      "dependencies": {
        "next": "^15.1.0",
        "react": "^19.0.0",
        "react-dom": "^19.0.0",
        "@repo/supabase": "workspace:*",
        "@repo/lib": "workspace:*"
      },
      "devDependencies": {
        "@repo/typescript-config": "workspace:*",
        "@types/node": "^22.0.0",
        "@types/react": "^19.0.0",
        "@types/react-dom": "^19.0.0",
        "typescript": "^5.7.0"
      }
    }
    ```

    2. Create next.config.ts:
    ```typescript
    import type { NextConfig } from 'next'
    import path from 'path'

    const nextConfig: NextConfig = {
      output: 'standalone',
      outputFileTracingRoot: path.join(__dirname, '../../'),
      transpilePackages: ['@repo/supabase', '@repo/lib'],
    }

    export default nextConfig
    ```

    3. Create tsconfig.json:
    ```json
    {
      "extends": "@repo/typescript-config/nextjs.json",
      "compilerOptions": {
        "baseUrl": ".",
        "paths": {
          "@/*": ["./*"]
        }
      },
      "include": ["next-env.d.ts", "**/*.ts", "**/*.tsx", ".next/types/**/*.ts"],
      "exclude": ["node_modules"]
    }
    ```

    4. Create app/layout.tsx:
    ```tsx
    import type { Metadata } from 'next'

    export const metadata: Metadata = {
      title: '외국인 구인구직',  // or '관리자 패널' for admin
      description: '한국어 가능한 외국인을 위한 구인구직 플랫폼',
    }

    export default function RootLayout({
      children,
    }: {
      children: React.ReactNode
    }) {
      return (
        <html lang="ko">
          <body>{children}</body>
        </html>
      )
    }
    ```

    5. Create app/page.tsx:
    ```tsx
    export default function Home() {
      return (
        <main>
          <h1>외국인 구인구직</h1>  // or '관리자 패널' for admin
          <p>Foundation setup complete.</p>
        </main>
      )
    }
    ```

    6. Create .env.example for each app:
    ```
    NEXT_PUBLIC_SUPABASE_URL=your_supabase_url
    NEXT_PUBLIC_SUPABASE_ANON_KEY=your_anon_key
    ```

    Note: apps/admin uses port 3001, apps/web uses port 3000
  </action>
  <verify>
    - apps/web and apps/admin directories exist with all files
    - Both apps have package.json with workspace dependencies
    - Port configuration: web=3000, admin=3001
  </verify>
  <done>Both Next.js applications created with correct configuration</done>
</task>
