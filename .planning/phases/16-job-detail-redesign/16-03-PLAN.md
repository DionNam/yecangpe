1. Add `generateMetadata` export to `apps/web/app/(main)/jobs/[slug]/page.tsx`:
   ```ts
   export async function generateMetadata({ params }: Props): Promise<Metadata> {
     const { slug } = await params
     const supabase = await createClient()
     const { data: job } = await supabase
       .from('job_posts')
       .select('title, company_name, content, company_logo_url')
       .eq('slug', slug)
       .eq('review_status', 'published')
       .single()

     if (!job) return {}

     const description = job.content?.substring(0, 160) || ''

     return {
       title: `${job.title} | ${job.company_name}`,
       description,
       openGraph: {
         title: `${job.title} - ${job.company_name}`,
         description,
         type: 'article',
         locale: 'ko_KR',
         siteName: 'HangulJobs',
         ...(job.company_logo_url && { images: [{ url: job.company_logo_url }] }),
       },
       twitter: {
         card: 'summary',
         title: `${job.title} - ${job.company_name}`,
         description,
       },
     }
   }
   ```

2. Add schema.org JobPosting JSON-LD to the page render:
   - Create a helper function `buildJobPostingSchema(job)` that returns the JSON-LD object:
     ```ts
     function buildJobPostingSchema(job: JobPost) {
       // Map job_type to schema.org employmentType
       const employmentTypeMap: Record<string, string> = {
         full_time: 'FULL_TIME',
         part_time: 'PART_TIME',
         contract: 'CONTRACTOR',
         freelance: 'CONTRACTOR',
         internship: 'INTERN',
         temporary: 'TEMPORARY',
       }

       // Map salary_period to schema.org unitText
       const salaryPeriodMap: Record<string, string> = {
         hourly: 'HOUR',
         daily: 'DAY',
         weekly: 'WEEK',
         monthly: 'MONTH',
         yearly: 'YEAR',
       }

       return {
         '@context': 'https://schema.org',
         '@type': 'JobPosting',
         title: job.title,
         description: job.content,
         datePosted: job.published_at || job.created_at,
         ...(job.expires_at && { validThrough: job.expires_at }),
         hiringOrganization: {
           '@type': 'Organization',
           name: job.company_name,
           ...(job.company_logo_url && { logo: job.company_logo_url }),
         },
         ...(job.job_type && { employmentType: employmentTypeMap[job.job_type] }),
         ...(job.work_location_type === 'remote' && { jobLocationType: 'TELECOMMUTE' }),
         ...(job.salary_min && {
           baseSalary: {
             '@type': 'MonetaryAmount',
             currency: job.salary_currency || 'KRW',
             value: {
               '@type': 'QuantitativeValue',
               minValue: job.salary_min,
               ...(job.salary_max && { maxValue: job.salary_max }),
               ...(job.salary_period && { unitText: salaryPeriodMap[job.salary_period] }),
             },
           },
         }),
       }
     }
     ```
   - Render the JSON-LD as a script tag at the top of the page return:
     ```tsx
     return (
       <>
         <script
           type="application/ld+json"
           dangerouslySetInnerHTML={{ __html: JSON.stringify(buildJobPostingSchema(job)) }}
         />
         <JobDetailPage ... />
       </>
     )
     ```
