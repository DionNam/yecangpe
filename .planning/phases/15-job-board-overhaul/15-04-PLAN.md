    2. Build query with all filters:
       ```typescript
       let query = supabase
         .from('job_posts')
         .select('*', { count: 'exact' })
         .eq('review_status', 'published')

       // Keyword search (full-text search on fts column)
       if (keyword) {
         query = query.textSearch('fts', keyword, { type: 'websearch' })
       }

       // Nationality filter (existing, keep unchanged)
       if (nationality && nationality !== 'all') {
         query = query.or(`target_nationality.eq.${nationality},target_nationality.eq.ANY`)
       }

       // Job type filter (multi-select with .in())
       const jobTypes = params.job_type?.split(',').filter(Boolean)
       if (jobTypes && jobTypes.length > 0) {
         query = query.in('job_type', jobTypes)
       }

       // Work location type filter (multi-select with .in())
       const locationTypes = params.location_type?.split(',').filter(Boolean)
       if (locationTypes && locationTypes.length > 0) {
         query = query.in('work_location_type', locationTypes)
       }

       // Country filter
       if (locationCountry && locationCountry !== 'all') {
         query = query.eq('work_location_country', locationCountry)
       }

       // Category filter (single-select)
       if (category && category !== 'all') {
         query = query.eq('category', category)
       }

       // Korean level filter
       if (koreanLevel && koreanLevel !== 'all') {
         query = query.eq('korean_level', koreanLevel)
       }

       // English level filter
       if (englishLevel && englishLevel !== 'all') {
         query = query.eq('english_level', englishLevel)
       }

       // Sorting
       if (sortBy === 'relevance' && keyword) {
         // When keyword search with relevance sort, Supabase handles via FTS ranking
         // textSearch already returns in relevance order, just don't override with published_at
       } else {
         query = query.order('published_at', { ascending: false })
       }
       ```
