    ```typescript
    const rawData = {

    ```

    ```typescript
    const result = seekerProfileSchema.safeParse(rawData)
    if (!result.success) {
      return { error: result.error.flatten().fieldErrors }
    }
    ```

    Upsert users table first (required for FK constraint):
    ```typescript
    const { error: userError } = await supabase
      .from('users')
      .upsert({
        id: user.id,
        email: user.email!,
        role: 'seeker',
      })
    if (userError) throw userError
    ```

    ```typescript
    const { error: profileError } = await supabase
      .from('seeker_profiles')
      .insert({
        user_id: user.id,
        ...result.data,
      })
    if (profileError) throw profileError
    ```

    ```typescript
    revalidatePath('/')
    redirect('/')
    ```

    Component structure:
    ```typescript
    export function SeekerForm() {
      const [isPending, startTransition] = useTransition()

      const form = useForm<SeekerProfileInput>({
        resolver: zodResolver(seekerProfileSchema),
        defaultValues: {
          nationality: undefined,
          topik_level: null,
          occupation: '',
          referral_source: '',
        },
      })

      const onSubmit = (data: SeekerProfileInput) => {
        startTransition(async () => {
          const formData = new FormData()
          formData.append('nationality', data.nationality)
          if (data.topik_level !== null && data.topik_level !== undefined) {
            formData.append('topik_level', String(data.topik_level))
          }
          if (data.occupation) formData.append('occupation', data.occupation)
          if (data.referral_source) formData.append('referral_source', data.referral_source)
          await createSeekerProfile(formData)
        })
      }

      return (
        <Form {...form}>
          <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
            {/* Fields */}
          </form>
        </Form>
      )
    }
    ```
